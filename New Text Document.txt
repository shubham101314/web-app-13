*/ name: CI-CD Pipeline

on:
  push:
    branches:
      - main

env:
  IMAGE_NAME: shubham101314/cake-app

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest

    steps:

    - name: Checkout Code
      uses: actions/checkout@v4

    # 🔐 AWS AUTH
    - name: Configure AWS Credentials
      uses: aws-actions/configure-aws-credentials@v4
      with:
        aws-access-key-id: ${{ secrets.AWS_ACCESS_KEY_ID }}
        aws-secret-access-key: ${{ secrets.AWS_SECRET_ACCESS_KEY }}
        aws-region: ap-south-1

    # 🔍 GET EC2 IP (STRICT + FILTERED)
    - name: Get EC2 Public IP
      id: ec2
      run: |
        IP=$(aws ec2 describe-instances \
          --filters "Name=tag:Name,Values=frontend-server" \
                    "Name=instance-state-name,Values=running" \
          --query "Reservations[].Instances[].PublicIpAddress" \
          --output text)

        echo "Fetched IP: $IP"

        if [ -z "$IP" ] || [ "$IP" = "None" ]; then
          echo "❌ ERROR: No running EC2 instance found with tag frontend-server"
          exit 1
        fi

        echo "EC2_IP=$IP" >> $GITHUB_ENV

    # 🐳 DOCKER LOGIN
    - name: Login to DockerHub
      uses: docker/login-action@v3
      with:
        username: ${{ secrets.DOCKER_USERNAME }}
        password: ${{ secrets.DOCKER_PASSWORD }}

    # 🏗 BUILD + PUSH
    - name: Build and Push Docker Image
      run: |
        docker build -t $IMAGE_NAME:latest .
        docker push $IMAGE_NAME:latest

    # 🚀 DEPLOY (ONLY IF IP EXISTS)
    - name: Deploy to EC2 via SSH
      if: ${{ env.EC2_IP != '' }}
      uses: appleboy/ssh-action@v1.0.3
      with:
        host: ${{ env.EC2_IP }}
        username: ec2-user
        key: ${{ secrets.EC2_SSH_KEY }}
        script: |
          echo "Connected to EC2"
          docker --version || (echo "Docker not installed" && exit 1)

          docker pull $IMAGE_NAME:latest
          docker stop frontend || true
          docker rm frontend || true
          docker run -d -p 5000:80 --name frontend $IMAGE_NAME:latest
